let activeCity = "scottsdale";
let filterDone = false;
let state = JSON.parse(localStorage.getItem("sitevisit_state") || "{}");

function save() { localStorage.setItem("sitevisit_state", JSON.stringify(state)); }

function switchCity(city) {
  activeCity = city;
  document.getElementById("tab-scottsdale").className = "tab" + (city === "scottsdale" ? " active-scottsdale" : "");
  document.getElementById("tab-sanjose").className = "tab" + (city === "sanjose" ? " active-sanjose" : "");
  filterDone = false;
  render();
}

function toggleFilter() {
  filterDone = !filterDone;
  render();
}

function toggleSection(id) {
  state["collapsed_" + id] = !state["collapsed_" + id];
  save(); render();
}

function toggleItem(key) {
  state[key] = !state[key];
  save(); render();
}

function openNote(key) {
  state["open_" + key] = !state["open_" + key];
  render();
}

function updateNote(key, val) {
  state["note_" + key] = val;
  save();
  // update preview without full re-render
  const el = document.getElementById("notepreview_" + key);
  if (el) el.textContent = val ? "📝 " + (val.length > 60 ? val.slice(0,60)+"…" : val) : "";
}

function countSec(sec, city) {
  if (sec.subsections) return sec.subsections.reduce((a,ss) => a + ss.items.length, 0);
  return sec.items.length;
}
function doneSec(sec, city) {
  if (sec.subsections) return sec.subsections.reduce((a,ss) =>
    a + ss.items.filter((_,i) => state["done_"+city+"_"+ss.id+"_"+i]).length, 0);
  return sec.items.filter((_,i) => state["done_"+city+"_"+sec.id+"_"+i]).length;
}

function render() {
  const city = data[activeCity];
  const ck = activeCity;

  // Progress
  const total = city.sections.reduce((a,s) => a + countSec(s,ck), 0);
  const done = city.sections.reduce((a,s) => a + doneSec(s,ck), 0);
  const pct = total ? Math.round(done/total*100) : 0;
  document.getElementById("progress-box").className = "progress-box " + ck;
  document.getElementById("progress-label").className = "progress-label " + ck;
  document.getElementById("progress-label").textContent = "Progress";
  document.getElementById("progress-count").textContent = done + " / " + total + " items";
  document.getElementById("progress-fill").className = "progress-fill " + ck;
  document.getElementById("progress-fill").style.width = pct + "%";

  // Filter btn
  const fb = document.getElementById("filter-btn");
  fb.className = "filter-btn" + (filterDone ? " active-" + ck : "");
  fb.textContent = filterDone ? "✓ Hiding done" : "Show all";

  // Sections
  const container = document.getElementById("sections-container");
  container.innerHTML = "";

  city.sections.forEach(sec => {
    const isCollapsed = state["collapsed_" + sec.id];
    const secDone = doneSec(sec, ck);
    const secTotal = countSec(sec, ck);

    const secEl = document.createElement("div");
    secEl.className = "section " + ck;

    // Header
    secEl.innerHTML = `
      <button class="sec-header ${ck}" onclick="toggleSection('${sec.id}')">
        <div class="sec-header-left">
          <span class="sec-icon">${sec.icon}</span>
          <span class="sec-title ${ck}">${sec.title}</span>
        </div>
        <div class="sec-header-right">
          <span class="sec-count">${secDone}/${secTotal}</span>
          <span class="sec-arrow ${ck}">${isCollapsed ? "▸" : "▾"}</span>
        </div>
      </button>`;

    if (!isCollapsed) {
      const body = document.createElement("div");
      body.className = "sec-body";

      const renderItem = (item, id, idx) => {
        const dk = "done_" + ck + "_" + id + "_" + idx;
        const nk = ck + "_" + id + "_" + idx;
        const isDone = state[dk];
        const noteVal = state["note_" + nk] || "";
        const isOpen = state["open_" + nk];
        if (filterDone && isDone) return "";
        return `
          <div class="item">
            <div class="item-row">
              <button class="checkbox ${isDone ? "done-"+ck : ""}" onclick="toggleItem('${dk}')">${isDone ? "✓" : ""}</button>
              <div style="flex:1">
                <span class="item-text ${isDone ? "done" : ""}">${item}</span>
                <div id="notepreview_${nk}" class="note-preview">${noteVal && !isOpen ? "📝 " + (noteVal.length>60?noteVal.slice(0,60)+"…":noteVal) : ""}</div>
                ${isOpen ? `<textarea oninput="updateNote('${nk}', this.value)" placeholder="Add your notes here...">${noteVal}</textarea>` : ""}
              </div>
              <button class="note-btn ${noteVal ? "has-note-"+ck : ""}" onclick="openNote('${nk}')">${isOpen ? "💾" : "📝"}</button>
            </div>
          </div>`;
      };

      if (sec.subsections) {
        sec.subsections.forEach(ss => {
          const anyVisible = ss.items.some((_,i) => !filterDone || !state["done_"+ck+"_"+ss.id+"_"+i]);
          if (!anyVisible) return;
          body.innerHTML += `<div class="subsec-header ${ck}">${ss.label}</div>`;
          ss.items.forEach((item, i) => { body.innerHTML += renderItem(item, ss.id, i); });
        });
      } else {
        sec.items.forEach((item, i) => { body.innerHTML += renderItem(item, sec.id, i); });
      }

      secEl.appendChild(body);
    }

    container.appendChild(secEl);
  });
}

render();
