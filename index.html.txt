var city='s', filterOn=false, state={}, pendingPhotoKey=null;
try{state=JSON.parse(localStorage.getItem('sv2026')||'{}');}catch(e){state={};}

function save(){
  try{
    localStorage.setItem('sv2026',JSON.stringify(state));
    checkStorage();
  }catch(e){
    alert('Storage full! Please export your notes and photos now to avoid losing data.');
  }
}

function checkStorage(){
  try{
    var used=new Blob([JSON.stringify(state)]).size;
    document.getElementById('storage-warn').style.display=used>3500000?'block':'none';
  }catch(e){}
}

function switchCity(c){city=c;filterOn=false;document.getElementById('tab-s').className='tab'+(c==='s'?' tab-s':'');document.getElementById('tab-j').className='tab'+(c==='j'?' tab-j':'');render();}
function toggleFilter(){filterOn=!filterOn;render();}
function toggleSec(id){state['col_'+id]=!state['col_'+id];save();render();}
function toggleItem(dk){state[dk]=!state[dk];save();render();}
function toggleNote(nk){state['open_'+nk]=!state['open_'+nk];render();}
function saveNote(nk,val){state['note_'+nk]=val;save();}
function togglePhotos(nk){state['showphotos_'+nk]=!state['showphotos_'+nk];render();}

function triggerPhoto(nk){
  pendingPhotoKey=nk;
  var inp=document.getElementById('photo-input');
  inp.value='';
  inp.click();
}

function handlePhoto(e){
  var file=e.target.files[0];
  if(!file||!pendingPhotoKey)return;
  var nk=pendingPhotoKey;
  var photos=state['photos_'+nk]||[];
  if(photos.length>=3){alert('Maximum 3 photos per item.');return;}
  var reader=new FileReader();
  reader.onload=function(ev){
    var img=new Image();
    img.onload=function(){
      var canvas=document.createElement('canvas');
      var max=800;
      var w=img.width,h=img.height;
      if(w>h){if(w>max){h=Math.round(h*max/w);w=max;}}else{if(h>max){w=Math.round(w*max/h);h=max;}}
      canvas.width=w;canvas.height=h;
      canvas.getContext('2d').drawImage(img,0,0,w,h);
      var compressed=canvas.toDataURL('image/jpeg',0.7);
      photos.push(compressed);
      state['photos_'+nk]=photos;
      state['showphotos_'+nk]=true;
      save();render();
    };
    img.src=ev.target.result;
  };
  reader.readAsDataURL(file);
}

function deletePhoto(nk,idx){
  var photos=state['photos_'+nk]||[];
  photos.splice(idx,1);
  state['photos_'+nk]=photos;
  save();render();
}

function itemCount(sec){if(sec.subsections)return sec.subsections.reduce(function(a,ss){return a+ss.items.length;},0);return sec.items.length;}
function doneCount(sec){var c=city;if(sec.subsections)return sec.subsections.reduce(function(a,ss){return a+ss.items.filter(function(_,i){return state['d_'+c+'_'+ss.id+'_'+i];}).length;},0);return sec.items.filter(function(_,i){return state['d_'+c+'_'+sec.id+'_'+i];}).length;}

function renderItem(item,id,i,c){
  var dk='d_'+c+'_'+id+'_'+i;
  var nk=c+'_'+id+'_'+i;
  var isDone=state[dk];
  var note=state['note_'+nk]||'';
  var isOpen=state['open_'+nk];
  var photos=state['photos_'+nk]||[];
  var showPhotos=state['showphotos_'+nk];
  if(filterOn&&isDone)return'';
  var cbCls='cb'+(isDone?' cb-'+c:'');
  var txtCls='item-txt'+(isDone?' item-done':'');
  var noteBtnCls='icon-btn'+(note?' icon-btn-'+c:'');
  var photoBtnCls='icon-btn'+(photos.length>0?' icon-btn-'+c:'');
  var out='<div class="item"><div class="item-row">';
  out+='<button class="'+cbCls+'" onclick="toggleItem(\''+dk+'\')">'+(isDone?'✓':'')+'</button>';
  out+='<div style="flex:1"><span class="'+txtCls+'">'+item+'</span>';
  if(note&&!isOpen)out+='<div class="note-pre">📝 '+(note.length>60?note.slice(0,60)+'…':note)+'</div>';
  if(isOpen)out+='<textarea onchange="saveNote(\''+nk+'\',this.value)" oninput="saveNote(\''+nk+'\',this.value)">'+escHtml(note)+'</textarea>';
  if(showPhotos&&photos.length>0){
    out+='<div class="photos-row">';
    photos.forEach(function(src,pi){
      out+='<div class="photo-wrap"><img src="'+src+'"><button class="photo-del" onclick="deletePhoto(\''+nk+'\','+pi+')">✕</button></div>';
    });
    if(photos.length<3)out+='<button class="add-photo-btn" onclick="triggerPhoto(\''+nk+'\')">📷</button>';
    out+='</div>';
  }
  out+='</div>';
  out+='<div class="action-btns">';
  out+='<button class="'+noteBtnCls+'" onclick="toggleNote(\''+nk+'\')" title="Note">'+(isOpen?'💾':'📝')+'</button>';
  out+='<button class="'+photoBtnCls+'" onclick="'+(showPhotos||photos.length===0?'triggerPhoto':'togglePhotos')+'(\''+nk+'\')" title="Photo">📷'+(photos.length>0?' <span style="font-size:10px;vertical-align:top;">'+photos.length+'</span>':'')+'</button>';
  out+='</div>';
  out+='</div></div>';
  return out;
}

function escHtml(s){return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');}

function buildSummary(){
  var out='';
  ['s','j'].forEach(function(c){
    var C=CITIES[c];
    out+=C.label.toUpperCase()+'\n'+'='.repeat(40)+'\n\n';
    C.sections.forEach(function(sec){
      var secOut='';
      function proc(items,id){items.forEach(function(item,i){var dk='d_'+c+'_'+id+'_'+i;var note=state['note_'+c+'_'+id+'_'+i]||'';var photos=state['photos_'+c+'_'+id+'_'+i]||[];if(state[dk]||note||photos.length){secOut+=(state[dk]?'  ✓ ':'  – ')+item+'\n';if(note)secOut+='    📝 '+note+'\n';if(photos.length)secOut+='    📷 '+photos.length+' photo(s) attached\n';}});}
      if(sec.subsections){sec.subsections.forEach(function(ss){proc(ss.items,ss.id);});}else{proc(sec.items,sec.id);}
      if(secOut)out+=sec.icon+' '+sec.title+'\n'+secOut+'\n';
    });
  });
  return out;
}

}
